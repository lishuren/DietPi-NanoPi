# DietPi-Automation
# This file allows you to automate the setup process of DietPi.
# Place this file on the boot partition of your SD card.

# ----- Global Options -----
AUTO_SETUP_AUTOMATED=1
AUTO_SETUP_GLOBAL_PASSWORD=dietpi
AUTO_SETUP_TIMEZONE=Asia/Shanghai
AUTO_SETUP_KEYBOARD_LAYOUT=us

# ----- Network Options -----
# Enable Ethernet (1=On)
AUTO_SETUP_NET_ETHERNET_ENABLED=1
# Enable WiFi (1=On)
AUTO_SETUP_NET_WIFI_ENABLED=0
# WiFi Credentials (if enabled above)
AUTO_SETUP_NET_WIFI_SSID=MyWiFiName
AUTO_SETUP_NET_WIFI_KEY=MyWiFiPassword

# ----- Software Install -----
# Install software automatically:
# 105 - OpenSSH Server (SSH access)
# 132 - Aria2 (Download manager)
# 85  - Nginx (Web server)
# 96  - Samba Server (File sharing)
# 89  - PHP (Web scripting for control panel)
AUTO_SETUP_INSTALL_SOFTWARE_ID=105 132 85 96 89

# ----- SSH Server -----
# 0=Dropbear (Lightweight), 1=OpenSSH (Full feature)
AUTO_SETUP_SSH_SERVER_INDEX=-1

# ----- Logging -----
# 0=Ramlog (Save SD card writes), 1=Full logging
AUTO_SETUP_LOGGING_INDEX=0
